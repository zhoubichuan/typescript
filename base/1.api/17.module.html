# 2.命名空间

命名空间可以用于组织代码，避免文件内命名冲突

# 2.1 使用

  • 在需要使用得地方放入
// type.d.ts
export namespace zoo {
  export class Dog {
    eat() {
      console.log("zoo dog")
    }
  }
}
export namespace home {
  export class Dog {
    eat() {
      console.log("home dog")
    }
  }
}

let dog_of_zoo = new zoo.Dog()
dog_of_zoo.eat()
let dog_of_home = new home.Dog()
dog_of_home.eat()
