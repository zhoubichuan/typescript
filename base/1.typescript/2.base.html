# 4.Enum

使⽤枚举我们可以定义⼀些带名字的常量

  USER, // 默认从0开始
  ADMIN,
  MANAGER,
}
// {0: "USER", 1: "ADMIN", 2: "MANAGER", USER: 0, ADMIN: 1, MANAGER: 2}

# 8.Never

任何类型的子类型，never 代表不会出现的值。不能把其他类型赋值给 never

function error(message: string): never {
  throw new Error("err")
}
function loop(): never {
  while (true) {}
}
function fn(x: number | string) {
  if (typeof x === "number") {
  } else if (typeof x === "string") {
  } else {
    console.log(x) //never
  }
}
